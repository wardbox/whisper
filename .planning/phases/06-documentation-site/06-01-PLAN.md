1. **Create `packages/docs/components/mdx-components.tsx`** with AutoTypeTable generator pointed at the library source:

   ```tsx
   import type { MDXComponents } from 'mdx/types';
   import defaultMdxComponents from 'fumadocs-ui/mdx';
   import { createGenerator, createFileSystemGeneratorCache } from 'fumadocs-typescript';
   import { AutoTypeTable, type AutoTypeTableProps } from 'fumadocs-typescript/ui';

   const generator = createGenerator({
     cache: createFileSystemGeneratorCache('.next/fumadocs-typescript'),
     tsconfigPath: '../whisper/tsconfig.json',
   });

   export function getMDXComponents(components?: MDXComponents): MDXComponents {
     return {
       ...defaultMdxComponents,
       ...components,
       AutoTypeTable: (props: Partial<AutoTypeTableProps>) => (
         <AutoTypeTable {...props} generator={generator} />
       ),
     };
   }
   ```

   Key decisions:
   - `tsconfigPath` points at the library's tsconfig so the TypeScript Compiler API resolves library types correctly.
   - Source `.ts` files (not `.d.ts`) are referenced in MDX so JSDoc comments are preserved in the rendered output.
   - The `.next/fumadocs-typescript` cache directory avoids recompilation on every page load.
