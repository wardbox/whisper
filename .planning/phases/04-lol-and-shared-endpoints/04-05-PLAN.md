Create `packages/whisper/src/lol/routing.test.ts` with compile-time type tests ensuring PlatformRoute/RegionalRoute constraints are enforced.

```typescript
import { describe, it } from 'vitest';
import type { WhisperClient } from '../core/client.js';
// Import all namespace objects
import { championMasteryV4 } from './champion-mastery-v4.js';
import { championV3 } from './champion-v3.js';
import { clashV1 } from './clash-v1.js';
import { leagueV4 } from './league-v4.js';
import { leagueExpV4 } from './league-exp-v4.js';
import { lolChallengesV1 } from './lol-challenges-v1.js';
import { lolStatusV4 } from './lol-status-v4.js';
import { matchV5 } from './match-v5.js';
import { lolRsoMatchV1 } from './lol-rso-match-v1.js';
import { spectatorV5 } from './spectator-v5.js';
import { summonerV4 } from './summoner-v4.js';
import { tournamentV5 } from './tournament-v5.js';
import { tournamentStubV5 } from './tournament-stub-v5.js';

declare const client: WhisperClient;

describe('route type enforcement', () => {
  it('platform-routed modules reject regional routes', () => {
    // @ts-expect-error -- regional route to platform-only endpoint
    summonerV4.getByPuuid(client, 'americas', 'puuid');
    // @ts-expect-error
    championMasteryV4.getByPuuid(client, 'europe', 'puuid');
    // @ts-expect-error
    championV3.getChampionRotations(client, 'asia');
    // @ts-expect-error
    clashV1.getPlayersByPuuid(client, 'sea', 'puuid');
    // @ts-expect-error
    leagueV4.getChallengerLeague(client, 'americas', 'RANKED_SOLO_5x5');
    // @ts-expect-error
    leagueExpV4.getEntries(client, 'europe', 'RANKED_SOLO_5x5', 'GOLD', 'I');
    // @ts-expect-error
    lolChallengesV1.getConfig(client, 'asia');
    // @ts-expect-error
    lolStatusV4.getStatus(client, 'sea');
    // @ts-expect-error
    spectatorV5.getCurrentGame(client, 'americas', 'puuid');
  });

  it('regional-routed modules reject platform routes', () => {
    // @ts-expect-error -- platform route to regional-only endpoint
    matchV5.getMatch(client, 'na1', 'matchId');
    // @ts-expect-error
    matchV5.getMatchIdsByPuuid(client, 'euw1', 'puuid');
    // @ts-expect-error
    lolRsoMatchV1.getMatch(client, 'kr', 'matchId');
    // @ts-expect-error
    tournamentV5.createProvider(client, 'na1', { region: 'NA', url: 'https://example.com' });
    // @ts-expect-error
    tournamentStubV5.createProvider(client, 'euw1', { region: 'EUW', url: 'https://example.com' });
  });
});
```

This file uses `@ts-expect-error` assertions. If the type constraints are correct, TypeScript will flag the error and `@ts-expect-error` will suppress it. If a module accidentally accepts the wrong route type, the `@ts-expect-error` will be unused and TypeScript will report an error -- catching the bug.
