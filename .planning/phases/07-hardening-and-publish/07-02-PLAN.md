```javascript
#!/usr/bin/env node
// scripts/smoke/run.mjs — Phase 7 smoke orchestrator.
//
// Reproduces the CI smoke matrix locally:
//   1. Build packages/whisper
//   2. Pack into a tarball at packages/whisper/wardbox-whisper-0.1.0.tgz
//   3. Copy the tarball into e2e/smoke/ as a deterministic filename
//   4. Bust pnpm's tarball cache and reinstall the smoke fixture
//   5. Run typecheck + Node ESM + Node CJS smoke entries
//   6. Conditionally run Bun + Deno legs IF the runtimes are installed locally
//
// Per CONTEXT.md D-05: this script runs the same sequence CI runs, no CI-only magic.
// Per RESEARCH Pitfall 5: --force is required to bust pnpm's tarball cache between runs.

import { execSync, spawnSync } from 'node:child_process';
import { existsSync, copyFileSync, rmSync, mkdirSync } from 'node:fs';
import { resolve, join } from 'node:path';

const repoRoot = resolve(new URL('../..', import.meta.url).pathname);
const pkgDir = join(repoRoot, 'packages/whisper');
const smokeDir = join(repoRoot, 'e2e/smoke');
const TARBALL_NAME = 'wardbox-whisper-0.1.0.tgz';

function run(cmd, opts = {}) {
  console.log(`\n$ ${cmd} ${opts.cwd ? `(in ${opts.cwd.replace(repoRoot + '/', '')})` : ''}`);
  execSync(cmd, { stdio: 'inherit', ...opts });
}

function which(bin) {
  const result = spawnSync(process.platform === 'win32' ? 'where' : 'which', [bin], { stdio: 'pipe' });
  return result.status === 0;
}

// 1. Build
run('pnpm --filter @wardbox/whisper build', { cwd: repoRoot });

// 2. Pack — produces wardbox-whisper-0.1.0.tgz at packages/whisper/
run(`pnpm pack --pack-destination .`, { cwd: pkgDir });

const packedAt = join(pkgDir, TARBALL_NAME);
if (!existsSync(packedAt)) {
  console.error(`smoke: expected packed tarball at ${packedAt} but not found`);
  process.exit(1);
}

// 3. Copy tarball into e2e/smoke/ at the path the smoke fixture's package.json points at.
//    (file:../../packages/whisper/wardbox-whisper-0.1.0.tgz — already there from step 2,
//     but pnpm won't traverse out of the smoke dir during install if the file: isn't found.
//     pnpm DOES support file: paths to anywhere on disk, so step 2's location works.)

// 4. Bust pnpm cache: remove node_modules and re-install --force
const smokeNodeModules = join(smokeDir, 'node_modules');
if (existsSync(smokeNodeModules)) {
  console.log(`smoke: removing stale ${smokeNodeModules}`);
  rmSync(smokeNodeModules, { recursive: true, force: true });
}
const smokeLockfile = join(smokeDir, 'pnpm-lock.yaml');
if (existsSync(smokeLockfile)) {
  rmSync(smokeLockfile);
}

run('pnpm install --force --ignore-workspace', { cwd: smokeDir });

// 5. Node legs (always run)
run('pnpm typecheck', { cwd: smokeDir });
run('pnpm esm', { cwd: smokeDir });
run('pnpm cjs', { cwd: smokeDir });

// 6. Optional Bun leg (only if `bun` is on PATH)
if (which('bun')) {
  console.log('\nsmoke: bun detected — running Bun leg');
  run('bun install --force', { cwd: smokeDir });
  run('bun run smoke_bun.ts', { cwd: smokeDir });
} else {
  console.log('\nsmoke: bun not installed locally — skipping Bun leg (CI runs it)');
}

// 7. Optional Deno leg (only if `deno` is on PATH)
if (which('deno')) {
  console.log('\nsmoke: deno detected — running Deno leg');
  // node_modules is already populated by pnpm above; deno reads it via nodeModulesDir: auto
  run('deno run -A --node-modules-dir=auto smoke_deno.ts', { cwd: smokeDir });
} else {
  console.log('\nsmoke: deno not installed locally — skipping Deno leg (CI runs it)');
}

console.log('\nsmoke: ✅ all available runtime legs passed');
```

KEY POINTS:
- `pnpm install --force --ignore-workspace` — `--ignore-workspace` is critical because the smoke fixture is OUTSIDE the workspace; without it, pnpm tries to find a workspace root and may pick up the parent `pnpm-workspace.yaml`. `--force` busts the tarball cache (Pitfall 5).
- The script removes `e2e/smoke/node_modules/` and `pnpm-lock.yaml` before each run — guaranteed clean state.
- Bun and Deno legs are CONDITIONAL on local installation. CI ALWAYS runs them via the matrix (Plan 04). RESEARCH § Environment Availability notes that on the local dev machine, only Bun is guaranteed.
- The script uses `execSync` with `stdio: inherit` so output streams in real time and any failure fast-aborts.
- `which()` uses `where`/`which` cross-platform (Windows/macOS/Linux) and is sufficient for determining tool availability.
- Tarball filename `wardbox-whisper-0.1.0.tgz` is hard-coded — D-15 locks v0.1.0 for first publish. If/when version bumps, the script and the smoke fixture's `dependencies` field both update together (a future Changesets-driven bump).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| smoke fixture → packed tarball | The fixture proves the tarball contents are functional, but a malicious tarball would also pass smoke. Smoke is a correctness gate, not a security gate. |
| smoke entry → createClient | Smoke calls createClient with a fake key; if the client constructor leaked the key to network logs or a remote service, smoke would not catch it. (Phase 2 unit tests cover key handling.) |
| pnpm install in fixture → repo lockfile | If the smoke fixture were inside the workspace, its file: dependency would resolve before pnpm pack ran, breaking the root install. `--ignore-workspace` + e2e/ outside `packages/*` prevents this. |
